    <script>
        // Configuración de tarifas
        const TARIFAS = {
            NORMAL: { ida: 3.50, vuelta: 3.50, completo: 7.00 },
            ESPECIAL: { ida: 2.50, vuelta: 2.50, completo: 5.00 }
        };

        const NOMBRES_PASAJERAS = ['Katherine', 'Fanny', 'Greta', 'María', 'Jesennia'];
        const INICIALES = ['K', 'F', 'G', 'M', 'J'];
        const ES_GRETA_ESPECIAL = [false, false, true, false, false]; // Solo Greta (índice 2) es especial

        // Estado
        let pasajeras = NOMBRES_PASAJERAS.map((nombre, i) => ({
            id: i + 1,
            nombre: nombre,
            inicial: INICIALES[i],
            esEspecial: ES_GRETA_ESPECIAL[i],
            tarifa: ES_GRETA_ESPECIAL[i] ? TARIFAS.ESPECIAL : TARIFAS.NORMAL,
            tipo: 'ninguno' // 'ninguno', 'ida', 'vuelta', 'completo'
        }));
        
        let registros = JSON.parse(localStorage.getItem('expresoRegistros')) || [];
        let registroAEliminar = null;

        // Inicialización
        document.addEventListener('DOMContentLoaded', () => {
            document.getElementById('fechaServicio').valueAsDate = new Date();
            renderizarPasajeras();
            actualizarTotales();
            renderizarRegistros();
        });

        function renderizarPasajeras() {
            const container = document.getElementById('passengerList');
            container.innerHTML = '';

            pasajeras.forEach((p, index) => {
                const item = document.createElement('div');
                const claseEspecial = p.esEspecial ? 'especial' : '';
                item.className = `passenger-item ${claseEspecial} ${p.tipo !== 'ninguno' ? 'active-' + p.tipo : ''}`;
                
                const tarifaText = p.esEspecial ? '? $2.50/$5.00' : '$3.50/$7.00';
                
                item.innerHTML = `
                    <div class="passenger-info">
                        <div class="passenger-avatar">${p.inicial}</div>
                        <div class="passenger-details">
                            <h4>${p.nombre} ${p.esEspecial ? '?' : ''}</h4>
                            <div class="tarifa-info">Tarifa: ${tarifaText}</div>
                            <span id="status-${index}">${obtenerTextoEstado(p.tipo)}</span>
                        </div>
                    </div>
                    <div class="passenger-toggle">
                        <button class="toggle-btn ${p.tipo === 'ida' ? 'active' : ''}" 
                                data-type="ida" onclick="cambiarTipo(${index}, 'ida', event)">
                            IDA<br>$${p.tarifa.ida.toFixed(2)}
                        </button>
                        <button class="toggle-btn ${p.tipo === 'vuelta' ? 'active' : ''}" 
                                data-type="vuelta" onclick="cambiarTipo(${index}, 'vuelta', event)">
                            VUELTA<br>$${p.tarifa.vuelta.toFixed(2)}
                        </button>
                        <button class="toggle-btn ${p.tipo === 'completo' ? 'active' : ''}" 
                                data-type="completo" onclick="cambiarTipo(${index}, 'completo', event)">
                            COMPLETO<br>$${p.tarifa.completo.toFixed(2)}
                        </button>
                    </div>
                `;
                
                container.appendChild(item);
            });

            calcularTotal();
        }

        function obtenerTextoEstado(tipo) {
            if (tipo === 'ninguno') return 'No viaja ?';
            if (tipo === 'ida') return 'Solo ida ??';
            if (tipo === 'vuelta') return 'Solo vuelta ??';
            if (tipo === 'completo') return 'Ida y vuelta ??';
        }

        function cambiarTipo(index, tipo, event) {
            event.stopPropagation();
            // Si ya está seleccionado este tipo, lo deseleccionamos (volvemos a 'ninguno')
            if (pasajeras[index].tipo === tipo) {
                pasajeras[index].tipo = 'ninguno';
            } else {
                pasajeras[index].tipo = tipo;
            }
            renderizarPasajeras();
        }

        function seleccionarTodas(tipo) {
            pasajeras.forEach(p => p.tipo = tipo);
            renderizarPasajeras();
            if (navigator.vibrate) navigator.vibrate(50);
        }

        function limpiarSeleccion() {
            pasajeras.forEach(p => p.tipo = 'ninguno');
            renderizarPasajeras();
        }

        function calcularTotal() {
            let total = 0;
            let countIda = 0;
            let countVuelta = 0;
            let countCompleto = 0;
            let countEspecial = 0;
            let detalles = [];

            pasajeras.forEach(p => {
                if (p.tipo === 'ida') { 
                    total += p.tarifa.ida; 
                    countIda++;
                    if (p.esEspecial) countEspecial++;
                }
                else if (p.tipo === 'vuelta') { 
                    total += p.tarifa.vuelta; 
                    countVuelta++;
                    if (p.esEspecial) countEspecial++;
                }
                else if (p.tipo === 'completo') { 
                    total += p.tarifa.completo; 
                    countCompleto++;
                    if (p.esEspecial) countEspecial++;
                }
            });

            const totalPasajeras = countIda + countVuelta + countCompleto;
            
            document.getElementById('calculatedTotal').textContent = total.toFixed(2);
            
            let breakdownText = `${totalPasajeras} pasajeras: `;
            if (countIda > 0) breakdownText += `${countIda} ida `;
            if (countVuelta > 0) breakdownText += `${countVuelta} vuelta `;
            if (countCompleto > 0) breakdownText += `${countCompleto} completo `;
            if (countEspecial > 0) breakdownText += `(${countEspecial} con tarifa especial)`;
            
            document.getElementById('totalBreakdown').textContent = totalPasajeras > 0 ? breakdownText : '0 pasajeras seleccionadas';
            
            return { total, countIda, countVuelta, countCompleto, countEspecial };
        }

        function guardarRegistro() {
            const fecha = document.getElementById('fechaServicio').value;
            if (!fecha) {
                alert('Por favor selecciona una fecha');
                return;
            }

            const pasajerasActivas = pasajeras.filter(p => p.tipo !== 'ninguno');
            if (pasajerasActivas.length === 0) {
                alert('Selecciona al menos una pasajera');
                return;
            }

            const calculo = calcularTotal();
            const registro = {
                id: Date.now(),
                fecha: fecha,
                fechaObj: new Date(fecha).toISOString(),
                pasajeras: JSON.parse(JSON.stringify(pasajeras)),
                resumen: {
                    total: calculo.total,
                    ida: calculo.countIda,
                    vuelta: calculo.countVuelta,
                    completo: calculo.countCompleto,
                    especiales: calculo.countEspecial
                },
                timestamp: new Date().toISOString()
            };

            registros.unshift(registro);
            localStorage.setItem('expresoRegistros', JSON.stringify(registros));

            // Resetear formulario
            limpiarSeleccion();
            actualizarTotales();
            renderizarRegistros();

            // Feedback táctil y visual
            if (navigator.vibrate) navigator.vibrate([50, 100, 50]);
            
            const btn = document.querySelector('.btn-primary');
            const originalText = btn.innerHTML;
            btn.innerHTML = '? ˇRegistro Guardado!';
            btn.style.background = '#10b981';
            setTimeout(() => {
                btn.innerHTML = originalText;
                btn.style.background = '';
            }, 2000);
        }

        function actualizarTotales() {
            const hoy = new Date().toISOString().split('T')[0];
            const inicioSemana = new Date();
            inicioSemana.setDate(inicioSemana.getDate() - inicioSemana.getDay());
            
            const totalHoy = registros
                .filter(r => r.fecha === hoy)
                .reduce((sum, r) => sum + r.resumen.total, 0);
            
            const totalSemana = registros
                .filter(r => new Date(r.fecha) >= inicioSemana)
                .reduce((sum, r) => sum + r.resumen.total, 0);

            const totalMes = registros.reduce((sum, r) => sum + r.resumen.total, 0);

            document.getElementById('totalHoy').textContent = `$${totalHoy.toFixed(2)}`;
            document.getElementById('totalSemana').textContent = `$${totalSemana.toFixed(2)}`;
            document.getElementById('totalMes').textContent = `$${totalMes.toFixed(2)}`;
            document.getElementById('totalViajes').textContent = registros.length;
        }

        function renderizarRegistros() {
            const container = document.getElementById('recordsList');
            
            if (registros.length === 0) {
                container.innerHTML = `
                    <div class="empty-state">
                        <div class="empty-state-icon">??</div>
                        <p>No hay registros aún</p>
                    </div>
                `;
                return;
            }

            container.innerHTML = registros.map(r => {
                const detalles = r.pasajeras
                    .filter(p => p.tipo !== 'ninguno')
                    .map(p => {
                        let clase = '';
                        let icono = '';
                        let monto = 0;
                        
                        if (p.tipo === 'ida') { 
                            clase = p.esEspecial ? 'badge-especial' : 'badge-ida'; 
                            icono = '??'; 
                            monto = p.tarifa.ida;
                        }
                        else if (p.tipo === 'vuelta') { 
                            clase = p.esEspecial ? 'badge-especial' : 'badge-vuelta'; 
                            icono = '??'; 
                            monto = p.tarifa.vuelta;
                        }
                        else if (p.tipo === 'completo') { 
                            clase = p.esEspecial ? 'badge-especial' : 'badge-completo'; 
                            icono = '??'; 
                            monto = p.tarifa.completo;
                        }
                        
                        const especialTag = p.esEspecial ? '? ' : '';
                        return `<span class="record-badge ${clase}">${icono} ${especialTag}${p.nombre} ($${monto.toFixed(2)})</span>`;
                    }).join('');

                return `
                    <div class="record-item">
                        <div class="record-header">
                            <span class="record-date">${formatearFecha(r.fecha)}</span>
                            <div style="display: flex; align-items: center;">
                                <span class="record-amount">$${r.resumen.total.toFixed(2)}</span>
                                <button class="delete-btn" onclick="eliminarRegistro(${r.id})">×</button>
                            </div>
                        </div>
                        <div class="record-passengers">
                            ${detalles}
                        </div>
                    </div>
                `;
            }).join('');
        }

        function formatearFecha(fechaStr) {
            const fecha = new Date(fechaStr + 'T00:00:00');
            const opciones = { weekday: 'short', day: 'numeric', month: 'short', year: '2-digit' };
            return fecha.toLocaleDateString('es-EC', opciones);
        }

        function eliminarRegistro(id) {
            registroAEliminar = id;
            document.getElementById('deleteModal').classList.add('active');
        }

        function confirmarEliminar() {
            if (registroAEliminar) {
                registros = registros.filter(r => r.id !== registroAEliminar);
                localStorage.setItem('expresoRegistros', JSON.stringify(registros));
                actualizarTotales();
                renderizarRegistros();
                registroAEliminar = null;
            }
            cerrarModal();
        }

        function cerrarModal() {
            document.getElementById('deleteModal').classList.remove('active');
            registroAEliminar = null;
        }

        function exportarExcel() {
            if (registros.length === 0) {
                alert('No hay registros para exportar');
                return;
            }

            let csv = 'Fecha,Katherine,Fanny,Greta (Especial),María,Jesennia,Total\n';
            
            registros.forEach(r => {
                const fila = [r.fecha];
                r.pasajeras.forEach(p => {
                    let texto = '';
                    let monto = 0;
                    if (p.tipo === 'ninguno') texto = 'No viaja';
                    else if (p.tipo === 'ida') {
                        monto = p.tarifa.ida;
                        texto = p.esEspecial ? `Ida Especial ($${monto.toFixed(2)})` : `Ida ($${monto.toFixed(2)})`;
                    }
                    else if (p.tipo === 'vuelta') {
                        monto = p.tarifa.vuelta;
                        texto = p.esEspecial ? `Vuelta Especial ($${monto.toFixed(2)})` : `Vuelta ($${monto.toFixed(2)})`;
                    }
                    else if (p.tipo === 'completo') {
                        monto = p.tarifa.completo;
                        texto = p.esEspecial ? `Completo Especial ($${monto.toFixed(2)})` : `Completo ($${monto.toFixed(2)})`;
                    }
                    fila.push(texto);
                });
                fila.push(r.resumen.total.toFixed(2));
                csv += fila.join(',') + '\n';
            });

            // Agregar resumen
            csv += '\nRESUMEN,,,,,\n';
            
            // Contar por tipo y tarifa
            let totalIdaNormal = 0, totalIdaEspecial = 0;
            let totalVueltaNormal = 0, totalVueltaEspecial = 0;
            let totalCompletoNormal = 0, totalCompletoEspecial = 0;
            
            registros.forEach(r => {
                r.pasajeras.forEach(p => {
                    if (p.tipo === 'ida') {
                        if (p.esEspecial) totalIdaEspecial++;
                        else totalIdaNormal++;
                    } else if (p.tipo === 'vuelta') {
                        if (p.esEspecial) totalVueltaEspecial++;
                        else totalVueltaNormal++;
                    } else if (p.tipo === 'completo') {
                        if (p.esEspecial) totalCompletoEspecial++;
                        else totalCompletoNormal++;
                    }
                });
            });
            
            csv += `Viajes Ida Normal (K,F,M,J),${totalIdaNormal},$${(totalIdaNormal * TARIFAS.NORMAL.ida).toFixed(2)}\n`;
            csv += `Viajes Ida Especial (Greta),${totalIdaEspecial},$${(totalIdaEspecial * TARIFAS.ESPECIAL.ida).toFixed(2)}\n`;
            csv += `Viajes Vuelta Normal (K,F,M,J),${totalVueltaNormal},$${(totalVueltaNormal * TARIFAS.NORMAL.vuelta).toFixed(2)}\n`;
            csv += `Viajes Vuelta Especial (Greta),${totalVueltaEspecial},$${(totalVueltaEspecial * TARIFAS.ESPECIAL.vuelta).toFixed(2)}\n`;
            csv += `Viajes Completo Normal (K,F,M,J),${totalCompletoNormal},$${(totalCompletoNormal * TARIFAS.NORMAL.completo).toFixed(2)}\n`;
            csv += `Viajes Completo Especial (Greta),${totalCompletoEspecial},$${(totalCompletoEspecial * TARIFAS.ESPECIAL.completo).toFixed(2)}\n`;
            csv += '\n';
            csv += `TOTAL GENERAL,,,$${registros.reduce((sum, r) => sum + r.resumen.total, 0).toFixed(2)}\n`;

            // Descargar
            const blob = new Blob(["\ufeff" + csv], { type: 'text/csv;charset=utf-8;' });
            const link = document.createElement('a');
            link.href = URL.createObjectURL(blob);
            link.download = `Expreso_Milagro_GYE_${new Date().toISOString().split('T')[0]}.csv`;
            link.click();

            if (navigator.vibrate) navigator.vibrate([50, 100, 50]);
        }

        // Cerrar modal al hacer clic fuera
        document.getElementById('deleteModal').addEventListener('click', (e) => {
            if (e.target.id === 'deleteModal') cerrarModal();
        });
